def initialize(atStr, data)
  Saklient::Util::validate_type(atStr, 'String')
  @_at = Saklient::Util::str2date(atStr)
  @_is_available = true
  v = nil
  v = data[:Out]
  if (v).nil?
    @_is_available = false
  else
    @_outgoing = v
  end
  v = data[:In]
  if (v).nil?
    @_is_available = false
  else
    @_incoming = v
  end
end
